\justifying{\textbf{Definition (Free Group $F_2$)}. The free group $F_2$ generated by two elements $\sigma, \tau$ is the group consisting of all finite products of $\sigma, \tau$ and their inverses $\sigma^{-1}, \tau^{-1}$. The identity in this group is the empty product (denoted $e$).}
\end{center}
